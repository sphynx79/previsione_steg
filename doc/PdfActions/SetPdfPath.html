def self.version
  files = Dir.glob("#{ctx.path}/STEG_*_#{ctx.data}_rev*.pdf")
  return "1" if files.empty?
  last_version = files.max_by { |x| x.scan(/rev\d/) }.match(%r{(?<version>rev\d)})
  return "1" if last_version.nil?
  last_version[:version][-1].to_i + 1
end
