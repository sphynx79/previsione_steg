def self.compila(previsione, sheet)
  previsione.each do |k, v|
    previsione[k] = v.each_slice(1).to_a
    column = column(k)
    ctx.workbook.Worksheets(sheet).Range("#{column}3:#{column}26").value = v.each_slice(1).to_a
  end
end
