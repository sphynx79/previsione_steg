def self.get_applica_somiglianza
  binding.pry
  unless ["SI", "NO"].include? applica_somiglianza
    ctx.fail_and_return!(
      <<~HEREDOC
        Controllare che nel file: Forecast.xlsm 
        Foglio: "Forecast V2"
        Ranking somiglianza: ci sia "SI" o "NO"
      HEREDOC
    )
  end
  applica_somiglianza
end
